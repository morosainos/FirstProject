(select tsLdidTwo from tListTripleStore triple 
inner join (select ldID from tListdata list1 where list1.ldValue ='521110') value on value.ldID = triple.tsLdidOne 
inner join (select ldId from tListdata list2 where list2.ldValue = 'NAICS/NACE') code on code.ldID = triple.tsLdidReason) sic 
on list3.ldID = sic.tsLdidTwo order by ldOrder
